#include <iostream>
#include <unordered_set>
using namespace std;

int main() {
    int m, n;

    cin >> m;
    int A[m];

    for(int i = 0; i < m; i++)
        cin >> A[i];

    cin >> n;
    int B[n];

    for(int i = 0; i < n; i++)
        cin >> B[i];

    unordered_set<int> s;

    for(int i = 0; i < m; i++)
        s.insert(A[i]);

    for(int i = 0; i < n; i++) {
        if(s.find(B[i]) != s.end())
            cout << B[i] << " ";
    }

    return 0;
}
